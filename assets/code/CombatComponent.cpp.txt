void UCombatComponent::RequestReloadAmmo(int32 RequestedAmount, E_AmmoType Type, UWeaponComponent* RequestingWeapon)
{
    // check reserve ammo type
    int32* AvailableAmmoPtr = ReserveAmmoMap.Find(Type);

    if (AvailableAmmoPtr)
    {
        int32& AvailableAmmo = *AvailableAmmoPtr;
        const int32 AmmoToTransfer = FMath::Min(RequestedAmount, AvailableAmmo);

        if (AmmoToTransfer > 0)
        {
            //AvailableAmmo -= AmmoToTransfer;

            RequestingWeapon->GrantReloadAmmo(AmmoToTransfer);
			RemoveReserveAmmo(Type, AmmoToTransfer);
        }
    }
}

void UCombatComponent::AddReserveAmmo(E_AmmoType AmmoType, int32 Amount)
{
    if (Amount <= 0) return;

    int32 CurrentAmmo = ReserveAmmoMap.FindOrAdd(AmmoType);

    int32 NewTotal = CurrentAmmo + Amount;
    ReserveAmmoMap.Add(AmmoType, NewTotal);

    //UE_LOG(LogTemp, Log, TEXT("Added %d of %s. New total: %d"), Amount, *AmmoType.ToString(), NewTotal);

    OnReserveAmmoChanged.Broadcast(AmmoType, NewTotal);
}

void UCombatComponent::UpdateReserveAmmoUI(E_AmmoType AmmoType)
{
	int32 CurrentTotal = GetReserveAmmo(AmmoType);

    	OnReserveAmmoChanged.Broadcast(AmmoType, CurrentTotal);
}

int32 UCombatComponent::GetReserveAmmo(E_AmmoType AmmoType)
{
    int32* CurrentAmmoPtr = ReserveAmmoMap.Find(AmmoType);
    return CurrentAmmoPtr ? *CurrentAmmoPtr : 0;
}

void UCombatComponent::RemoveReserveAmmo(E_AmmoType AmmoType, int32 Amount)
{
    if (Amount <= 0) return;
    int32* CurrentAmmoPtr = ReserveAmmoMap.Find(AmmoType);
    if (CurrentAmmoPtr)
    {
        int32& CurrentAmmo = *CurrentAmmoPtr;
        int32 NewTotal = FMath::Max(0, CurrentAmmo - Amount);
        ReserveAmmoMap.Add(AmmoType, NewTotal);
        //UE_LOG(LogTemp, Log, TEXT("Removed %d of %s. New total: %d"), Amount, *AmmoType.ToString(), NewTotal);
        OnReserveAmmoChanged.Broadcast(AmmoType, NewTotal);
    }
}

void UCombatComponent::HandleSecondaryInputAction(bool bIsPressed)
{
    const bool bHasRightWeapon = IsValid(RightHandWeaponComponent) && RightHandWeaponComponent->GetCurrentWeapon();
    const bool bHasLeftWeapon = IsValid(LeftHandWeaponComponent) && LeftHandWeaponComponent->GetCurrentWeapon();

    if (bIsPressed)
    {
        if (bHasLeftWeapon)
        {
            UE_LOG(LogTemp, Warning, TEXT("Fire Left"));
            LeftHandWeaponComponent->Fire(); 
        }
        else if (bHasRightWeapon)
        {
            UE_LOG(LogTemp, Warning, TEXT("ADS"));
            RightHandWeaponComponent->SetAiming(true);
        }
    }
    else
    {
        if (!bHasLeftWeapon && bHasRightWeapon)
        {
            RightHandWeaponComponent->SetAiming(false);
        }
    }
}

void UCombatComponent::SetWeaponReference(UWeaponComponent* WeaponComponent, E_WeaponHand WeaponHand)
{
    if (WeaponHand==E_WeaponHand::EWH_RightHand)
    {
        RightHandWeaponComponent = WeaponComponent;
    }
    else if (WeaponHand==E_WeaponHand::EWH_LeftHand)
    {
        LeftHandWeaponComponent = WeaponComponent;
    }
}
